NEW TYPE DEFINITIONS:
---------------------

// Slim chord recommendation (from AI)
interface AIChordRecommendationSlim {
  chordName: string;      // "Cmaj7", "Dm7", "G7"
  rootNote: string;       // "C", "D", "G"
  rationale: string;      // Why recommended
  genreContext: string;   // "Jazz, Blues"
  difficulty: number;     // 1-10
}

// Enriched chord recommendation
interface AIChordRecommendation extends AIChordRecommendationSlim {
  voicings: ChordVoicing[];  // From chord database
  notes: string[];           // Chord tones
  quality: string;           // "major7", "minor7", etc.
}

// Progression recommendation
interface AIChordProgressionRecommendation {
  name: string;              // "ii-V-I in C"
  chords: string[];          // ["Dm7", "G7", "Cmaj7"]
  romanNumerals: string[];   // ["ii7", "V7", "Imaj7"]
  rationale: string;
  genreContext: string;
  difficulty: number;
}
